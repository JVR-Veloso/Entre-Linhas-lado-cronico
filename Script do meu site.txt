    /* ======= Dados de exemplo ======= */
    const STORIES = [
      {
        id:1, title:'A insígnia esquecida', date:'2025-07-10', category:'ficcao', tags:['aventura','escotismo'], excerpt:'Um jovem encontra uma insígnia antiga que muda a maneira como ele enxerga a comunidade.',
        content:`<p><strong>Sinopse</strong>: Em uma tarde quente, João encontra uma insígnia coberta de pó...</p>
        <p>O texto completo deve vir aqui. Substitua este parágrafo pelo seu conto. Mantenha a formatação em HTML ou converta de Markdown se preferir.</p>`
      },
      {
        id:2, title:'Trilhas e decisões — capítulo 1', date:'2025-06-01', category:'serie', tags:['série','capítulo 1'], excerpt:'Primeiro capítulo de uma série sobre escolhas morais em acampamentos.',
        content:<p>Capítulo 1 — As escolhas pela manhã...</p>
      },
      {
        id:3, title:'Reflexão sobre liderança', date:'2025-04-20', category:'reflexao', tags:['ensaio','liderança'], excerpt:'Uma breve reflexão sobre o que significa liderar em pequenos grupos.',
        content:<p>O que é liderança senão a arte de servir? Neste texto, exploro...</p>
      }
    ];

    /* ======= Estado simples ======= */
    let currentStoryIndex = 0;

    function formatDate(d){
      // recebe YYYY-MM-DD
      try{const dt=new Date(d);return dt.toLocaleDateString('pt-BR',{day:'2-digit',month:'short',year:'numeric'})}catch(e){return d}
    }

    function renderList(){
      const list = document.getElementById('storiesList');
      const search = (document.getElementById('searchInput').value || '').toLowerCase().trim();
      const cat = document.getElementById('filterCat').value;
      const filtered = STORIES.filter(s=>{
        if(cat && s.category!==cat) return false;
        if(!search) return true;
        return (s.title+s.excerpt+(s.tags||[]).join(' ')).toLowerCase().includes(search);
      });

      list.innerHTML = filtered.map((s,idx)=>`
        <article class="story">
          <h3>\${s.title}</h3>
          <div class="meta">\${formatDate(s.date)} • \${s.tags.join(', ')}
          </div>
          <p class="excerpt">\${s.excerpt}</p>
          <div class="btn-row">
            <a class="btn" href="#" onclick="openStory(\${s.id});return false">Ler</a>
            <a class="btn" href="#" onclick="alert('Download de PDF (simulação)');return false">Baixar</a>
          </div>
        </article>
      `).join('');

      // destaque: primeiro item do filtro
      const latest = document.getElementById('latest');
      if(filtered.length){
        const top = filtered[0];
        latest.innerHTML = <div style=\"display:flex;gap:12px;align-items:center\"><div style=\"width:64px;height:64px;border-radius:8px;background:linear-gradient(135deg,var(--accent),var(--accent-2));color:#fff;display:flex;align-items:center;justify-content:center;font-weight:700\">S</div><div><strong>\${top.title}</strong><div class=\"meta\"> \${formatDate(top.date)} • \${top.tags.join(', ')}</div></div></div>;
      } else { latest.innerHTML = '<div class="muted">Nenhuma história encontrada</div>' }
    }

    function openStory(id){
      const idx = STORIES.findIndex(s=>s.id===id);
      if(idx===-1) return;
      currentStoryIndex = idx;
      const s = STORIES[idx];
      const container = document.getElementById('storyContent');
      container.innerHTML = <h1>\${s.title}</h1><div class="meta">\${formatDate(s.date)} • \${s.tags.join(', ')}</div><hr style=\"margin:12px 0\"> + s.content + <hr><p class=\"muted\">Fonte: Projeto Insígnia BP — ${s.title}</p>;
      document.getElementById('overlay').classList.add('open');
      renderNavPrevNext();
      // Atualizar url (sem reload)
      history.pushState({story:id}, '', #story-${id});
    }

    function closeOverlay(e){
      document.getElementById('overlay').classList.remove('open');
      history.pushState({}, '', location.pathname);
    }

    function renderNavPrevNext(){
      const box = document.getElementById('navPrevNext');
      const prev = STORIES[currentStoryIndex-1];
      const next = STORIES[currentStoryIndex+1];
      box.innerHTML = `
        <div style=\"display:flex;gap:8px\">${prev?<a class=\"btn\" href=\"#\" onclick=\"openStory(${prev.id});return false\">← ${prev.title}</a>:''}${next?<a class=\"btn\" href=\"#\" onclick=\"openStory(${next.id});return false\">${next.title} →</a>:''}</div>
      `
    }

    function shareStory(e){
      e.preventDefault();
      const s = STORIES[currentStoryIndex];
      const url = location.origin + location.pathname + #story-${s.id};
      if(navigator.share){
        navigator.share({title:s.title,text:s.excerpt,url}).catch(()=>alert('Compartilhamento cancelado'))
      } else {
        prompt('Copie o link:', url)
      }
    }

    // Mobile menu simple
    function toggleMenu(){
      const ul = document.querySelector('nav ul');
      ul.style.display = ul.style.display==='flex'?'none':'flex';
    }

    // Ao carregar, renderiza
    document.addEventListener('DOMContentLoaded', ()=>{
      renderList();
      // Se abriu com #story-{id}
      if(location.hash.startsWith('#story-')){
        const id = parseInt(location.hash.replace('#story-',''));
        if(id) openStory(id);
      }

      // Navegação simples por hash (scroll para seção)
      document.querySelectorAll('.navlink').forEach(a=>{
        a.addEventListener('click', (ev)=>{ev.preventDefault();const t=ev.target.dataset.target;document.getElementById(t).scrollIntoView({behavior:'smooth'})})
      });

      // Voltar/avançar do histórico fecha overlay quando não tiver story
      window.addEventListener('popstate', (ev)=>{
        if(!location.hash.startsWith('#story-')) document.getElementById('overlay').classList.remove('open');
      })
    })

    // Para adicionar histórias: edite a constante STORIES no topo.
